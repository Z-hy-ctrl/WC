/*****************************************************************头文件**********************************************************/
#include<stdio.h>
#include<stdlib.h>
#include<windows.h>
#include<time.h>
/*****************************************************************函数声明*********************************************************/
void muban();
void Pos(int x, int y);
/*****************************************************************自定义函数*******************************************************/
void Pos(int x, int y)//设置光标位置，从哪里开始输出
{
    COORD pos;//表示一个字符在控制台屏幕上的坐标，左上角(0,0)
    HANDLE hOutput;
    pos.X = x;
    pos.Y = y;
    hOutput = GetStdHandle(STD_OUTPUT_HANDLE);//返回标准的输入、输出或错误的设备的句柄，也就是获得输入、输出/错误的屏幕缓冲区的句柄
    SetConsoleCursorPosition(hOutput, pos);
}
void muban()
{	
	int i;
	for(i=0;i<=60;i+=2)//方块水平方向占两个字节
	{
       Pos(i,0);
	   printf("▇");//上行
	   Pos(i,26);
	   printf("▇");//下行
	}
	for(i=0;i<=25;i+=1)//方块竖直方向占1个字节
	{
       Pos(0,i);//左列
	   printf("▇");
	   Pos(60,i);//右列
	   printf("▇");
	}
}
/*******************************************************************主函数************************************************************/
int main()
{
    muban();
	return 0;
}
